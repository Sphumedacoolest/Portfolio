Calculator.
App

A fully functional calculator built with vanilla JavaScript. Features standard arithmetic, a scientific mode with trigonometric and logarithmic functions, keyboard input support, and a calculation history log.

// Evaluate expression safely function calculate() { let expr = expression .replace(/÷/g, '/') .replace(/×/g, '*') .replace(/−/g, '-') .replace(/π/g, Math.PI); const result = eval(expr); pushHistory(expression, result); return result; }
calculator.js — scientific ops JS
// Scientific functions function sciOp(op) { const n = parseFloat(current); const ops = { sin: () => Math.sin(n * Math.PI / 180), cos: () => Math.cos(n * Math.PI / 180), tan: () => Math.tan(n * Math.PI / 180), log: () => Math.log10(n), sqrt: () => Math.sqrt(n), pow2: () => n ** 2, inv: () => 1 / n, }; current = String(ops[op]()); updateDisplay(); }

// Keyboard event listener document.addEventListener('keydown', e => { if ('0123456789.'.includes(e.key)) inputNum(e.key); else if (e.key === 'Enter') calculate(); else if (e.key === 'Backspace') backspace(); else if (e.key === 'Escape') clearCalc(); else if (['+','-','*','/'].includes(e.key)) inputOp(e.key); });

Features

Toggle between standard and scientific with sin, cos, tan, log, square root and more.
